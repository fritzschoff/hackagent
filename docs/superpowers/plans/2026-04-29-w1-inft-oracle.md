# W1 — ERC-7857 INFT oracle implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the missing primitive of ERC-7857: every authorized INFT transfer re-encrypts the agent's memory blob to the new owner, with on-chain verification matching 0G's reference byte layout.

**Architecture:** New `AgentINFTVerifier` contract implements 0G's `IERC7857DataVerifier` byte-for-byte and adds the on-chain oracle-attestation check the reference left as TODO. Rewritten `AgentINFT` exposes `transferWithProof(to, tokenId, proof)`, a delegation table for receiver-proxy flows, and a `memoryReencrypted` stale flag for raw `transferFrom` bypasses. `AgentBids` and `AgentMerger` thread proofs through. A Vercel-hosted oracle service (Redis-backed AES-128 keys + 0G Storage anchoring + secp256k1 ECDSA + ECIES-secp256k1 wrap) prepares proofs in a 2-phase commit (`prepare-transfer` → `confirm-transfer`).

**Tech Stack:** Solidity 0.8.28 (Foundry), TypeScript (Next.js 16 App Router on Vercel), `@0glabs/0g-ts-sdk`, viem, ethers v6, ioredis, secp256k1 ECIES via `@noble/curves` + `@noble/ciphers`, OpenZeppelin v5.

**Spec:** [docs/superpowers/specs/2026-04-28-agent-identity-package-design.md](../specs/2026-04-28-agent-identity-package-design.md), W1 sections.

**Branch:** `feat/w1-inft-oracle` (already created).

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.28;

import {IERC7857DataVerifier, PreimageProofOutput, TransferValidityProofOutput} from "./IERC7857DataVerifier.sol";

/// @notice Implements 0G's IERC7857DataVerifier. Proof byte layout matches
/// 0G/0gfoundation/0g-agent-nft@eip-7857-draft contracts/verifiers/Verifier.sol.
/// Adds the on-chain oracle-attestation check the reference left as TODO.
contract AgentINFTVerifier is IERC7857DataVerifier {
    address public immutable EXPECTED_ORACLE;

    /// keccak256(nonce) -> used; replay protection.
    mapping(bytes32 => bool) public usedNonces;

    error InvalidProofLength();
    error InvalidOracleSignature();
    error NonceReplay();
    error WrongFlags();

    constructor(address oracle_) {
        require(oracle_ != address(0), "oracle zero");
        EXPECTED_ORACLE = oracle_;
    }

    function expectedOracle() external view returns (address) {
        return EXPECTED_ORACLE;
    }

    /// Mint proof layout: [1 flags | 65 sig | 32 dataHash | 48 nonce] = 146B
    /// Signature is EIP-191 over keccak256("inft-mint-v1" || dataHash || nonce).
    function verifyPreimage(bytes[] calldata _proofs)
        external returns (PreimageProofOutput[] memory)
    {
        PreimageProofOutput[] memory outputs = new PreimageProofOutput[](_proofs.length);
        for (uint256 i = 0; i < _proofs.length; i++) {
            bytes calldata p = _proofs[i];
            if (p.length != 146) revert InvalidProofLength();
            // Mint variant uses flags 0x00 (bit7=TEE=0, bit6=isPrivate=0).
            if (uint8(p[0]) != 0x00) revert WrongFlags();

            bytes32 r;
            bytes32 s;
            uint8 v;
            assembly {
                r := calldataload(add(p.offset, 1))
                s := calldataload(add(p.offset, 33))
                v := byte(0, calldataload(add(p.offset, 65)))
            }
            bytes32 dataHash = bytes32(p[66:98]);
            bytes memory nonce = p[98:146];

            bytes32 nonceKey = keccak256(nonce);
            if (usedNonces[nonceKey]) revert NonceReplay();

            bytes32 messageHash = keccak256(
                abi.encodePacked("inft-mint-v1", dataHash, nonce)
            );
            bytes32 prefixed = keccak256(
                abi.encodePacked("\x19Ethereum Signed Message:\n32", messageHash)
            );
            address recovered = ecrecover(prefixed, v, r, s);
            if (recovered != EXPECTED_ORACLE) revert InvalidOracleSignature();

            usedNonces[nonceKey] = true;
            outputs[i] = PreimageProofOutput({dataHash: dataHash, isValid: true});
        }
        return outputs;
    }

    /// Transfer proof layout: see AgentINFTVerifier verifyTransferValidity comment.
    /// Implementation in Task 5.
    function verifyTransferValidity(bytes[] calldata)
        external pure returns (TransferValidityProofOutput[] memory)
    {
        revert("not implemented");
    }
}
```

```solidity
    bytes32 oldHash,
    bytes32 newHash,
    bytes16 sealedKey,
    bytes memory nonce,
    string memory newUri,
    uint256 receiverPk,
    uint256 tokenId
) internal returns (bytes memory) {
    // Layout (private TEE flavor):
    // [0]    flags (0x40 = isPrivate=1, TEE=0)
    // [1..66) accessibility sig (65B over keccak256(newHash || oldHash || nonce) EIP-191)
    // [66..114) nonce (48B)
    // [114..146) newDataHash
    // [146..178) oldDataHash
    // [178..194) sealedKey (16B)
    // [194..227) ephemeralPubkey (33B compressed)
    // [227..239) ivWrap (12B)
    // [239..255) wrapTag (16B)
    // [255..257) newUriLength (uint16 BE)
    // [257..]   newUri (UTF-8) || oracleAttestation(65B)
    bytes1 flags = 0x40;

    bytes32 accessHash = keccak256(abi.encodePacked(newHash, oldHash, nonce));
    bytes memory accessSig = _signEip191(receiverPk, accessHash);

    // tests we don't actually verify these — only oracle attestation is checked.
    bytes memory ephemeralPubkey = abi.encodePacked(bytes33(uint264(0x02) << 256));

    bytes memory ephemeral = hex"0279BE667EF9DCBBAC55A06295CE870B07029BFCDB2DCE28D959F2815B16F81798";
```

```solidity
/// Transfer proof layout (private TEE flavor):
///   [0]      flags (bit7=TEE=0, bit6=isPrivate=1)
///   [1..66)  accessibility sig (65B over keccak256(newHash||oldHash||nonce) EIP-191)
///   [66..114) nonce (48B)
///   [114..146) newDataHash
///   [146..178) oldDataHash
///   [178..194) sealedKey (16B)
///   [194..227) ephemeralPubkey (33B compressed)
///   [227..239) ivWrap (12B)
///   [239..255) wrapTag (16B)
///   [255..257) newUriLen (uint16 BE)
///   [257..257+L) newUri (UTF-8)
///   [257+L..) oracleAttestation (65B sig over
///       keccak256(tokenId||oldHash||newHash||sealedKey||keccak256(newUri)||nonce))
function verifyTransferValidity(bytes[] calldata _proofs)
    external returns (TransferValidityProofOutput[] memory)
{
    TransferValidityProofOutput[] memory outputs =
        new TransferValidityProofOutput[](_proofs.length);
    for (uint256 i = 0; i < _proofs.length; i++) {
        outputs[i] = _verifyOne(_proofs[i]);
    }
    return outputs;
}

function _verifyOne(bytes calldata p)
    internal returns (TransferValidityProofOutput memory out)
{
    if (p.length < 257 + 65) revert InvalidProofLength();
    if (uint8(p[0]) != 0x40) revert WrongFlags();

    // accessibility sig (65B at [1..66))
    bytes32 accR;
    bytes32 accS;
    uint8 accV;
    assembly {
        accR := calldataload(add(p.offset, 1))
        accS := calldataload(add(p.offset, 33))
        accV := byte(0, calldataload(add(p.offset, 65)))
    }

    bytes memory nonce = p[66:114];
    bytes32 newDataHash = bytes32(p[114:146]);
    bytes32 oldDataHash = bytes32(p[146:178]);
    bytes16 sealedKey = bytes16(p[178:194]);

    // newUriLen at [255..257)
    uint16 uriLen = (uint16(uint8(p[255])) << 8) | uint16(uint8(p[256]));
    if (p.length != 257 + uint256(uriLen) + 65) revert InvalidProofLength();
    bytes memory newUri = p[257:257 + uriLen];

    // oracle attestation sig (last 65B)
    uint256 attestStart = 257 + uint256(uriLen);
    bytes32 attR;
    bytes32 attS;
    uint8 attV;
    assembly {
        attR := calldataload(add(p.offset, attestStart))
        attS := calldataload(add(add(p.offset, attestStart), 32))
        attV := byte(0, calldataload(add(add(p.offset, attestStart), 64)))
    }

    // Replay protection (per-nonce)
    bytes32 nonceKey = keccak256(nonce);
    if (usedNonces[nonceKey]) revert NonceReplay();

    // Recover receiver from accessibility sig
    bytes32 accessMsg = keccak256(abi.encodePacked(newDataHash, oldDataHash, nonce));
    bytes32 accessPrefixed = keccak256(
        abi.encodePacked("\x19Ethereum Signed Message:\n32", accessMsg)
    );
    address receiver = ecrecover(accessPrefixed, accV, accR, accS);
    if (receiver == address(0)) revert InvalidProofLength();

    // Recover oracle from attestation sig — this is the part 0G's reference
    // verifier left as `// TODO: verify TEE's signature`. We ship it.
    // tokenId is recovered via the AgentINFT contract calling us with tokenId
    // already bound off-band; for the verifier we accept that the attestation
    // is bound to the token by the oracle's signing convention. The on-chain
    // contract (AgentINFT) cross-checks oldDataHash == encryptedMemoryRoot
    // before trusting the proof, so any mismatch on tokenId is caught there.
    bytes32 attestMsg = keccak256(
        abi.encodePacked(
            // We can't easily get tokenId here; AgentINFT enforces oldHash
            // binding. The oracle includes tokenId in its sig but we re-derive
            // it from a wrapper structure in the new-uri tail by convention.
            // For test parity we sign over the same fields the test does.
```

**STOP — design correction needed before implementation.**

I noticed mid-write that the verifier doesn't have access to `tokenId` independently — it comes through the calldata only. Two options:

(a) Embed `tokenId` inside the proof bytes (extend layout: prepend a 32-byte tokenId field after flags). The verifier signs over the embedded value; AgentINFT reads it from calldata too and cross-checks.

(b) Have AgentINFT call a 2-arg variant `verifyTransferValidity(uint256 tokenId, bytes[] proofs)` so the verifier can include tokenId in the attestation hash without trusting calldata.

Option (b) deviates from the IERC7857DataVerifier interface. Option (a) keeps interface compatibility.

**Decision: option (a).** Prepend tokenId to the proof at offset 1..33, shift everything else by 32. Updated layout:

```
[0]        flags
[1..33)    tokenId (uint256)
[33..98)   accessibility sig (65B)
[98..146)  nonce (48B)
[146..178) newDataHash
[178..210) oldDataHash
[210..226) sealedKey
[226..259) ephemeralPubkey (33B)
[259..271) ivWrap (12B)
[271..287) wrapTag (16B)
[287..289) newUriLen
[289..289+L) newUri
[289+L..289+L+65) oracleAttestation
```

```solidity
        string memory baseTokenURI_,

        address oracle_
```

```solidity
    if (!out[0].isValid) revert InvalidTransferProof();
    if (out[0].oldDataHash != encryptedMemoryRoot[tokenId]) revert OldRootMismatch();

    bool directReceiverMatch = (out[0].receiver == to);
    bool delegationMatch;
    {
        Delegation memory d = delegations[to][tokenId];
        delegationMatch = (d.oracle == out[0].receiver
            && d.oracle == ORACLE
            && d.expiresAt > block.timestamp);
    }
    if (!directReceiverMatch && !delegationMatch) {
        if (delegations[to][tokenId].expiresAt > 0
            && delegations[to][tokenId].expiresAt <= block.timestamp) {
            revert DelegationExpired();
        }
        revert UndelegatedReceiver();
    }

    // Parse newUri from proof tail (offsets 287..289 = uriLen, then uri).
    uint16 uriLen = (uint16(uint8(proof[287])) << 8) | uint16(uint8(proof[288]));
    string memory newUri = string(proof[289:289 + uriLen]);
```

```typescript
import { hkdf } from "@noble/hashes/hkdf";
import { sha256 } from "@noble/hashes/sha2";
import { gcm } from "@noble/ciphers/aes";
import { hexToBytes, bytesToHex } from "@noble/curves/abstract/utils";
import { getRedis } from "@/lib/redis";

const KEK_SALT = new TextEncoder().encode("inft-kek-v1");

function kek(tokenId: bigint): Uint8Array {
  const pk = process.env.INFT_ORACLE_PK;
  if (!pk) throw new Error("INFT_ORACLE_PK missing");
  const skBytes = hexToBytes(pk.startsWith("0x") ? pk.slice(2) : pk);
  return hkdf(sha256, skBytes, KEK_SALT,
    new TextEncoder().encode(`tokenId:${tokenId.toString()}`), 32);
}

export async function storeKey(tokenId: bigint, aesKey: Uint8Array): Promise<void> {
  if (aesKey.length !== 16) throw new Error("aesKey must be 16 bytes");
  const k = kek(tokenId);
  const iv = crypto.getRandomValues(new Uint8Array(12));
  const ct = gcm(k, iv).encrypt(aesKey);
  const blob = new Uint8Array(iv.length + ct.length);
  blob.set(iv); blob.set(ct, iv.length);
  await getRedis().set(`inft:key:${tokenId}`, bytesToHex(blob));
}

export async function loadKey(tokenId: bigint): Promise<Uint8Array | null> {
  const hex = await getRedis().get(`inft:key:${tokenId}`);
  if (!hex) return null;
  const blob = hexToBytes(hex);
  const iv = blob.slice(0, 12);
  const ct = blob.slice(12);
  return gcm(kek(tokenId), iv).decrypt(ct);
}

export async function storePending(
  tokenId: bigint, nonceHex: string, aesKey: Uint8Array
): Promise<void> {
  const k = kek(tokenId);
  const iv = crypto.getRandomValues(new Uint8Array(12));
  const ct = gcm(k, iv).encrypt(aesKey);
  const blob = new Uint8Array(iv.length + ct.length);
  blob.set(iv); blob.set(ct, iv.length);
  await getRedis().set(
    `inft:pending:${tokenId}:${nonceHex}`, bytesToHex(blob), "EX", 86400
  );
}

export async function commitPending(tokenId: bigint, nonceHex: string): Promise<boolean> {
  const r = getRedis();
  const blob = await r.get(`inft:pending:${tokenId}:${nonceHex}`);
  if (!blob) return false;
  await r.set(`inft:key:${tokenId}`, blob);
  await r.del(`inft:pending:${tokenId}:${nonceHex}`);
  await r.incr(`inft:meta:${tokenId}:rotations`);
  return true;
}

export async function rotations(tokenId: bigint): Promise<number> {
  const v = await getRedis().get(`inft:meta:${tokenId}:rotations`);
  return v ? Number(v) : 0;
}
```

```typescript
import { gcm } from "@noble/ciphers/aes";
import { secp256k1 } from "@noble/curves/secp256k1";
import { hkdf } from "@noble/hashes/hkdf";
import { sha256 } from "@noble/hashes/sha2";
import { keccak_256 } from "@noble/hashes/sha3";
import { bytesToHex, hexToBytes } from "@noble/curves/abstract/utils";
import { Wallet } from "ethers";
import { writeBlob } from "@/lib/zg-storage-internal"; // exported helper, see Task 24

export type OraclePlaintext = Record<string, unknown>;

function oracleSk(): Uint8Array {
  const pk = process.env.INFT_ORACLE_PK;
  if (!pk) throw new Error("INFT_ORACLE_PK missing");
  return hexToBytes(pk.startsWith("0x") ? pk.slice(2) : pk);
}

export function oracleAddress(): `0x${string}` {
  const pk = process.env.INFT_ORACLE_PK!;
  return new Wallet(pk.startsWith("0x") ? pk : `0x${pk}`).address as `0x${string}`;
}

export function aesKeyFresh(): Uint8Array {
  return crypto.getRandomValues(new Uint8Array(16));
}

export function encryptBlob(plaintext: OraclePlaintext, key: Uint8Array): Uint8Array {
  const iv = crypto.getRandomValues(new Uint8Array(12));
  const json = new TextEncoder().encode(JSON.stringify(plaintext));
  const ct = gcm(key, iv).encrypt(json);
  const out = new Uint8Array(iv.length + ct.length);
  out.set(iv); out.set(ct, iv.length);
  return out;
}

export function decryptBlob(ciphertext: Uint8Array, key: Uint8Array): OraclePlaintext {
  const iv = ciphertext.slice(0, 12);
  const ct = ciphertext.slice(12);
  const pt = gcm(key, iv).decrypt(ct);
  return JSON.parse(new TextDecoder().decode(pt));
}

export function eciesWrap(aesKey: Uint8Array, recipientPubkey: Uint8Array)
  : { ephemeralPub: Uint8Array; iv: Uint8Array; sealedKey: Uint8Array; tag: Uint8Array }
{
  const ephSk = secp256k1.utils.randomPrivateKey();
  const ephPub = secp256k1.getPublicKey(ephSk, true); // compressed 33B
  const shared = secp256k1.getSharedSecret(ephSk, recipientPubkey, true); // 33B (0x02/0x03 prefix)
  const km = hkdf(sha256, shared.slice(1), new Uint8Array(),
    new TextEncoder().encode("inft-key-wrap-v1"), 32);
  const iv = crypto.getRandomValues(new Uint8Array(12));
  const ctTag = gcm(km.slice(0, 16), iv).encrypt(aesKey);
  const sealedKey = ctTag.slice(0, 16);  // first 16B of ct
  const tag = ctTag.slice(16);            // last 16B
  return { ephemeralPub: ephPub, iv, sealedKey, tag };
}

export function recoverPubkeyFromEip191(messageHash: Uint8Array, sigHex: `0x${string}`)
  : Uint8Array
{
  const sig = hexToBytes(sigHex.slice(2));
  const r = sig.slice(0, 32);
  const s = sig.slice(32, 64);
  const v = sig[64];
  const rec = v === 27 ? 0 : v === 28 ? 1 : v;
  const prefixed = keccak_256(new Uint8Array([
    ...new TextEncoder().encode("\x19Ethereum Signed Message:\n32"),
    ...messageHash,
  ]));
  const sigWithRec = new Uint8Array([...r, ...s]);
  // @noble/secp256k1 expects compact sig + recovery
  return secp256k1.Signature.fromCompact(sigWithRec)
    .addRecoveryBit(rec)
    .recoverPublicKey(prefixed)
    .toRawBytes(true);
}

export function signEip191(messageHash: Uint8Array): Uint8Array {
  const sk = oracleSk();
  const prefixed = keccak_256(new Uint8Array([
    ...new TextEncoder().encode("\x19Ethereum Signed Message:\n32"),
    ...messageHash,
  ]));
  const sig = secp256k1.sign(prefixed, sk);
  const compact = sig.toCompactRawBytes();
  const v = sig.recovery! + 27;
  return new Uint8Array([...compact, v]);
}

export function buildMintProof(dataHash: Uint8Array, nonce: Uint8Array): Uint8Array {
  // [1 flags | 65 sig | 32 dataHash | 48 nonce] = 146B
  const messageHash = keccak_256(new Uint8Array([
    ...new TextEncoder().encode("inft-mint-v1"),
    ...dataHash, ...nonce,
  ]));
  const sig = signEip191(messageHash);
  return new Uint8Array([0x00, ...sig, ...dataHash, ...nonce]);
}

export type TransferProofInput = {
  tokenId: bigint;
  oldRoot: Uint8Array;       // 32B
  newRoot: Uint8Array;       // 32B
  sealedKey: Uint8Array;     // 16B (from eciesWrap)
  ephemeralPub: Uint8Array;  // 33B
  ivWrap: Uint8Array;        // 12B
  wrapTag: Uint8Array;       // 16B
  newUri: string;            // UTF-8
  nonce: Uint8Array;         // 48B
  receiverSig: Uint8Array;   // 65B (oracle signs in delegation path)
};

export function buildTransferProof(i: TransferProofInput): Uint8Array {
  const uriBytes = new TextEncoder().encode(i.newUri);
  if (uriBytes.length > 0xFFFF) throw new Error("newUri too long");
  const uriLen = new Uint8Array([uriBytes.length >> 8, uriBytes.length & 0xff]);

  const tokenIdBytes = new Uint8Array(32);
  let n = i.tokenId;
  for (let j = 31; j >= 0; j--) { tokenIdBytes[j] = Number(n & 0xffn); n >>= 8n; }

  // Oracle attestation signature
  const attestMsg = keccak_256(new Uint8Array([
    ...tokenIdBytes,
    ...i.oldRoot, ...i.newRoot, ...i.sealedKey,
    ...keccak_256(uriBytes),
    ...i.nonce,
  ]));
  const oracleSig = signEip191(attestMsg);

  return new Uint8Array([
    0x40,                           // flags
    ...tokenIdBytes,                // 32
    ...i.receiverSig,               // 65 (oracle signs in delegation path)
    ...i.nonce,                     // 48
    ...i.newRoot,                   // 32
    ...i.oldRoot,                   // 32
    ...i.sealedKey,                 // 16
    ...i.ephemeralPub,              // 33
    ...i.ivWrap,                    // 12
    ...i.wrapTag,                   // 16
    ...uriLen,                      // 2
    ...uriBytes,                    // L
    ...oracleSig,                   // 65
  ]);
}

export function buildAccessSig(newRoot: Uint8Array, oldRoot: Uint8Array, nonce: Uint8Array)
  : Uint8Array
{
  const msg = keccak_256(new Uint8Array([...newRoot, ...oldRoot, ...nonce]));
  return signEip191(msg);
}

export async function anchorBlob(ciphertext: Uint8Array): Promise<{root: `0x${string}`; uri: string; txHash: `0x${string}`}> {
  const res = await writeBlob(ciphertext);
  if (!res || !res.anchored) throw new Error("0G anchor failed");
  return {
    root: res.rootHash as `0x${string}`,
    uri: `og://${res.rootHash}`,
    txHash: res.txHash as `0x${string}`,
  };
}
```
